Modelica definition

model Calibrated "Correlations with adjustment factors"
  extends Species(
    redeclare replaceable package Data = FCSys.Characteristics.H2O.Ionomer (b_v
          =[1], specVolPow={-1,0}),
    Xi=k_Xi*Data.Xi(T),
    F=k_F*Data.F(T),
    R=k_R*Data.R(T));

  parameter Q.NumberAbsolute k_Xi(final nominal=1) = 1 
    "Adjustment factor for fusivity (kΞ)";
  parameter Q.NumberAbsolute k_F(final nominal=1) = 1 
    "Adjustment factor for fluidity (kF)";
  parameter Q.NumberAbsolute k_R(final nominal=1) = 1 
    "Adjustment factor for thermal resistivity (kR)";

end Calibrated;
