FCSys.Characteristics.'e-'.Graphite.Polynomial.F

∫f()·dx evaluated at x with zero integration constant

Information

By definition, the partial derivative of this function with respect to x (with a constant) is f().

Extends from Modelica.Icons.Function (Icon for functions).

function F 
  "∫f()·dx evaluated at x with zero integration constant"
  extends Modelica.Icons.Function;

  input Real x "Argument";
  input Real a[:] "Coefficients";
  input Real n=0 "Power associated with the first term (before integral)";

  output Real F "Integral";

algorithm 
  F := f(
    x,
    a .* {if n + i == 0 then ln(x) else 1/(n + i) for i in 1:size(a, 1)},
    n + 1);
  // Note:  The derivative annotation is not used since f() is not
  // the direct and complete derivative of this function.

end F;

function f "Polynomial function which accepts non-integer powers"
  extends Modelica.Icons.Function;

  input Real u "Argument";
  input Real a[:] "Coefficients";
  input Real n=0 "Power of the first coefficient";
  output Real y "Result";

algorithm 
  y := sum(a[i]*u^(n + i - 1) for i in 1:size(a, 1));
end f;

function df 
  "Derivative of f()"
  annotation(derivative=d2f);
  extends Modelica.Icons.Function;

  input Real x "Argument";
  input Real a[:] "Coefficients";
  input Real n=0 "Power associated with the first term (before derivative)";
  input Real dx "Derivative of argument";
  input Real da[size(a, 1)] "Derivatives of coefficients";

  output Real df "Derivative";

algorithm 
  df := f(
    x,
    a={(n + i - 1)*a[i] for i in 1:size(a, 1)},
    n=n - 1)*dx + f(
    x,
    da,
    n);
end df;

function d2f 
  "Derivative of df()"
  extends Modelica.Icons.Function;

  input Real x "Argument";
  input Real a[:] "Coefficients";
  input Real n=0 "Power associated with the first term (before derivative)";
  input Real dx "Derivative of argument";
  input Real da[size(a, 1)] "Derivatives of coefficients";
  input Real d2x "Second derivative of argument";
  input Real d2a[size(a, 1)] "Second derivatives of coefficients";

  output Real d2f "Second derivative";

algorithm 
  d2f := sum(f(
    x,
    {a[i]*(n + i - 1)*(n + i - 2)*dx^2,(n + i - 1)*(2*da[i]*dx + a[i]*d2x),d2a[
      i]},
    n + i - 3) for i in 1:size(a, 1));

end d2f;
