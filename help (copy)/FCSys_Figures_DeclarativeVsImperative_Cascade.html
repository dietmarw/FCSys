model DeclarativeA "First circuit in declarative formalism"
  extends BaseClasses.Parameters;
  extends FCSys.BaseClasses.Icons.Blocks.Continuous;
  Connectors.RealInput vIn;
  Connectors.RealOutput vOut;
  Modelica.Electrical.Analog.Sources.SignalVoltage voltageSource;
  Modelica.Electrical.Analog.Basic.Resistor res1(R=R1);
  Modelica.Electrical.Analog.Basic.Capacitor cap(C=C);
  Modelica.Electrical.Analog.Basic.Ground ground;
  Modelica.Electrical.Analog.Sensors.VoltageSensor voltageSensor;

equation 
  connect(cap.n, ground.p);
  connect(cap.p, res1.n);
  connect(voltageSource.p, res1.p);
  connect(vIn, voltageSource.v);
  connect(voltageSource.n, ground.p);
  connect(vOut, voltageSensor.v);
  connect(voltageSensor.n, ground.p);
  connect(voltageSensor.p, res1.n);
end DeclarativeA;

model ImperativeA "First circuit in imperative formalism"
  extends BaseClasses.Parameters;
  extends FCSys.BaseClasses.Icons.Blocks.Continuous;
  Connectors.RealInput vIn;
  Connectors.RealOutput vOut;

model ImperativeAB 
  "Cascaded first and second circuits in imperative formalism"
  extends BaseClasses.Parameters;
  extends FCSys.BaseClasses.Icons.Blocks.Continuous;
  Connectors.RealInput vIn;
  Connectors.RealOutput vOut;
  Modelica.Blocks.Continuous.Integrator cap(k=1/C);
  Modelica.Blocks.Math.Gain res1(k=1/R1);
  Modelica.Blocks.Math.Gain res2(k=1/R2);
  Modelica.Blocks.Math.Gain res3(k=R3);
  Modelica.Blocks.Math.Add KVL1(k2=-1, k1=1);
  Modelica.Blocks.Math.Add KCL(k2=-1, k1=1);
  Modelica.Blocks.Math.Add KVL2(k2=-1);

equation 
  connect(res3.y, vOut);
  connect(KVL1.u1, vIn);
  connect(cap.y, KVL1.u2);
  connect(KCL.u1, res1.y);
  connect(KCL.y, cap.u);
  connect(KVL1.y, res1.u);
  connect(KVL2.y, res2.u);
  connect(res2.y, res3.u);
  connect(KCL.u2, res2.y);
  connect(KVL2.u1, cap.y);

  connect(KVL2.u2, res3.y);
end ImperativeAB;
