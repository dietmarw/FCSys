FCSys.Conditions.FacePair.Species

Conditions for a pair of Face connectors

Information


  
If the source of an internal specification is redeclared to a block besides
  Modelica.Blocks.Sources.Constant,
  then the related condition must be redeclared as well.  For example, use:

  redeclare Conditions.FacePair.Material.Pressure material(redeclare Modelica.Blocks.Sources.Ramp source)

  rather than simply:

  material(redeclare Modelica.Blocks.Sources.Ramp source)
  

  

Extends from FCSys.Conditions.BaseClasses.Icons.Double (Icon for a two-connector boundary condition).

Modelica definition


model Species 
  "
Conditions for a pair of Face connectors
"

  extends FCSys.Conditions.BaseClasses.Icons.Double;

  replaceable Material.Pressure material(source(k(start=U.atm))) constrainedby 
    Normal.BaseClasses.PartialCondition "Material";
  replaceable Transverse.Velocity transverse1(final orientation=Orientation.preceding,
      source(k(start=0))) constrainedby Transverse.BaseClasses.PartialCondition
    "
1st transverse
";
  replaceable Transverse.Velocity transverse2(final orientation=Orientation.following,
      source(k(start=0))) constrainedby Transverse.BaseClasses.PartialCondition
    "
2nd transverse
";
  replaceable Thermal.Temperature thermal(source(k(start=298.15*U.K))) 
    constrainedby Thermal.BaseClasses.PartialCondition "Thermal";
  // Note:  In Dymola 7.4, the value of k must be specified here instead
  // of at the lower level (e.g., Thermal.Temperature) so that the source
  // subcomponent can be replaced by blocks that don't contain the
  // parameter k.

  Connectors.Face negative 
    "Negative-side single-species connector for linear momentum and heat";

  Connectors.Face positive 
    "Positive-side single-species connector for linear momentum and heat";

  Connectors.RealInputBus u "Input bus for values of specified conditions";

  Connectors.RealOutputBus y "Output bus of measurements";

equation 
  // Normal
  connect(negative, normal.negative);
  connect(normal.positive, positive);
  connect(u.normal, normal.u);
  connect(normal.y, y.normal);

  // 1st transverse
  connect(negative, transverse1.negative);
  connect(transverse1.positive, positive);
  connect(u.transverse1, transverse1.u);
  connect(transverse1.y, y.transverse1);

  // 2nd transverse
  connect(negative, transverse2.negative);
  connect(transverse2.positive, positive);
  connect(u.transverse2, transverse2.u);
  connect(transverse2.y, y.transverse2);

  // Thermal
  connect(negative, thermal.negative);
  connect(thermal.positive, positive);
  connect(u.thermal, thermal.u);
  connect(thermal.y, y.thermal);
end Species;
