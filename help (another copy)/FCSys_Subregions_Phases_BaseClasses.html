FCSys.Subregions.Phases.BaseClasses.NullPhase

Model for a phase with no species or reactions

Information

If one of the species has setTemp = true, then
    initTemp should be set to false.
    Likewise, if one of the species has setVelX = true,
    setVelY = true, or setVelZ = true, then
    initVelX, initVelY, or initVelZ should
    be set to false (respectively).


    
Assumptions:
           
    
           
  1. The factors that may cause anisotropic transport (k)
          are common to all of the species within the phase.
    2. 




Notes:
    
  
  • The x-axis component of linear momentum is included by default.  At least one component must be included.

Modelica definition


model NullPhase "Model for a phase with no species or reactions"
  // extends FCSys.BaseClasses.Icons.Names.Middle;

  // Geometric parameters
  parameter Q.NumberAbsolute k[Axis](
    each min=Modelica.Constants.small,
    each final nominal=1) = {1,1,1} if n_spec > 0 
    "
Areal fill factor (k)
";

  // Assumptions
  parameter Boolean reduceVel=true if n_spec > 0 
    "Same velocity for all species";
  parameter Boolean reduceTemp=true if n_spec > 0 
    "Same temperature for all species";

  // Initialization
  parameter Boolean initVelX=true if n_spec > 0 "Initialize the x component";
  parameter Boolean initVelY=true if n_spec > 0 "Initialize the y component";
  parameter Boolean initVelZ=true if n_spec > 0 "Initialize the z component";
  parameter Q.Velocity phi_IC[Axis]={0,0,0} if n_spec > 0 
    "
Initial velocity (φIC)
";
  // This is always enabled in the dialog since it's used as a guess value.
  parameter Boolean initTemp=true if n_spec > 0 "Initialize";

  parameter Q.TemperatureAbsolute T_IC(nominal=298.15*U.K, start=environment.T)
    if n_spec > 0 "
Initial temperature (TIC)
";
  // This is always enabled in the dialog since it's used as a guess value.

  Connectors.ChemicalIOBus chemical if n_spec > 0 
    "Bus of chemical connectors for the species";

  Connectors.InertAmagat inert(final n_lin=n_lin) if n_spec > 0;
  Connectors.FaceBus xNegative if n_spec > 0 "Negative face along the x axis";
  Connectors.FaceBus xPositive if n_spec > 0 "Positive face along the x axis";
  Connectors.FaceBus yNegative if n_spec > 0 "Negative face along the y axis";
  Connectors.FaceBus yPositive if n_spec > 0 "Positive face along the y axis";
  Connectors.FaceBus zNegative if n_spec > 0 "Negative face along the z axis";
  Connectors.FaceBus zPositive if n_spec > 0 "Positive face along the z axis";

  outer parameter Q.Length L[Axis] if n_spec > 0 "Length";
  outer parameter Q.Area A[Axis] if n_spec > 0 "Cross-sectional area";
  // Note:  These must be public in Dymola 7.4, so HideResult is set true instead.

protected 
  final inner parameter Q.Length Lstar_trans[:]=k .* A ./ L if n_spec > 0 
    "Effective cross-sectional area per length";
  outer parameter Boolean inclLin[Axis] 
    "true, if each component of linear momentum is included";
  outer parameter Integer n_lin "Number of components of linear momentum";
  outer parameter Integer cartAxes[:] 
    "Cartesian-axis indices of the components of linear momentum";

  parameter Integer n_spec "Number of species";

  PhaseBoundary phaseBoundary if n_spec > 0 "Phase boundary";
  // This component is conditional because if two or more empty phases
  // (without any species included) were connected within a subregion, there
  // would be a mathematical singularity.

  Connectors.InertInternal common(
    n_lin=n_lin,
    final inclMechanical=reduceVel,
    final inclThermal=reduceTemp,
    mechanical(phi(each stateSelect=StateSelect.prefer, final start=phi_IC[
            cartAxes])),
    thermal(T(
        stateSelect=StateSelect.prefer,
        final start=T_IC,
        final fixed=initTemp))) if n_spec > 0 and (reduceVel or reduceTemp) 
    "Internal connector to directly couple velocities and/or temperatures";

  outer Conditions.Environment environment "Environmental conditions";

equation 
  // Inert interactions
  connect(phaseBoundary.inertA, inert);
end NullPhase;
