partial block DiscreteBlock "Base class of discrete blocks"
  extends FCSys.BaseClasses.Icons.Blocks.Discrete;
  parameter SI.Time samplePeriod(min=100*Modelica.Constants.eps, start=0.1) 
    "Sample period of species";
  parameter SI.Time startTime=0 "First sample time instant";

protected 
  output Boolean sampleTrigger "true, if sample time instant";
  output Boolean firstTrigger "Rising edge signals first sample instant";

equation 
  sampleTrigger = sample(startTime, samplePeriod);
  when sampleTrigger then
    firstTrigger = time <= startTime + samplePeriod/2;
  end when;

end DiscreteBlock;
