Test one flow plate to the other

Information

Extends from Modelica.Icons.Example (Icon for runnable examples).

Parameters

TypeNameDefaultDescription
Geometry

Modelica definition

model FPToFP "Test one flow plate to the other"
  import FCSys.Subregions.Phases;
  extends Modelica.Icons.Example;

  parameter Q.Length L_y[:]=fill(1*U.m/1, 1) 
    "Lengths along the channel (Ly)";
  parameter Q.Length L_z[:]=fill(5*U.mm/1, 1) 
    "Lengths across the channel (Lz)";
  final parameter Integer n_y=size(L_y, 1) 
    "Number of regions along the channel";
  final parameter Integer n_z=size(L_z, 1) 
    "Number of regions across the channel";
  inner FCSys.BCs.Environment environment(analysis=true);
  AnFPs.AnFP anFP(final L_y=L_y, final L_z=L_z);

  AnGDLs.AnGDL anGDL(final L_y=L_y, final L_z=L_z);

  AnCLs.AnCL anCL(final L_y=L_y, final L_z=L_z);

  PEMs.PEM pEM(final L_y=L_y, final L_z=L_z);

  CaCLs.CaCL caCL(final L_y=L_y, final L_z=L_z);

  CaGDLs.CaGDL caGDL(final L_y=L_y, final L_z=L_z);

  CaFPs.CaFP caFP(final L_y=L_y, final L_z=L_z);

  FCSys.BCs.FaceBus.SubregionClosed bC1[n_y, n_z](each graphite(
      inclC=true,
      'incle-'=true,
      C(entropySpec(k=environment.T)),
      'e-'(entropySpec(k=environment.T))));

  FCSys.BCs.FaceBus.SubregionClosed bC2[n_y, n_z](each graphite(
      inclC=true,
      'incle-'=true,
      C(entropySpec(k=environment.T)),
      'e-'(
        entropySpec(k=environment.T),
        redeclare FCSys.BCs.Face.Material.Pressure material,
        materialSpec(k(start=1*U.atm)))));
  FCSys.BCs.FaceBus.SubregionClosed bC3[anFP.n_x, n_z](each gas(
      inclH2=true,
      inclH2O=true,
      H2(entropySpec(k=environment.T)),
      H2O(entropySpec(k=environment.T))));

  FCSys.BCs.FaceBus.SubregionClosed bC4[anFP.n_x, n_z](each gas(
      inclH2=true,
      inclH2O=true,
      H2(entropySpec(k=environment.T)),
      H2O(entropySpec(k=environment.T))));

  FCSys.BCs.FaceBus.SubregionClosed bC5[caFP.n_x, n_z](each gas(
      inclH2O=true,
      inclN2=true,
      inclO2=true,
      N2(entropySpec(k=environment.T)),
      O2(entropySpec(k=environment.T)),
      H2O(entropySpec(k=environment.T))));
  FCSys.BCs.FaceBus.SubregionClosed bC6[caFP.n_x, n_z](each gas(
      inclH2O=true,
      inclN2=true,
      inclO2=true,
      N2(entropySpec(k=environment.T)),
      O2(entropySpec(k=environment.T)),
      H2O(entropySpec(k=environment.T))));

equation 
  connect(bC1.face, anFP.xNegative);
  connect(anFP.xPositive, anGDL.xNegative);
  connect(anGDL.xPositive, anCL.xNegative);

  connect(anCL.xPositive, pEM.xNegative);

  connect(pEM.xPositive, caCL.xNegative);

  connect(caCL.xPositive, caGDL.xNegative);

  connect(caGDL.xPositive, caFP.xNegative);

  connect(caFP.xPositive, bC2.face);

  connect(bC3.face, anFP.yNegative);
  connect(bC4.face, anFP.yPositive);
  connect(bC5.face, caFP.yNegative);
  connect(bC6.face, caFP.yPositive);
end FPToFP;

model CLToCL "Test one catalyst layer to the other"
  import FCSys.Subregions.Phases;
  extends Modelica.Icons.Example;

  parameter Q.Length L_y[:]=fill(1*U.m/1, 1) 
    "Lengths along the channel (Ly)";
  parameter Q.Length L_z[:]=fill(5*U.mm/1, 1) 
    "Lengths across the channel (Lz)";
  final parameter Integer n_y=size(L_y, 1) 
    "Number of regions along the channel";
  final parameter Integer n_z=size(L_z, 1) 
    "Number of regions across the channel";

  AnCLs.AnCL anCL(final L_y=L_y, final L_z=L_z);

  PEMs.PEM pEM(final L_y=L_y, final L_z=L_z);

  CaCLs.CaCL caCL(final L_y=L_y, final L_z=L_z);

  FCSys.BCs.FaceBus.SubregionFlow bC1[n_y, n_z](
    each gas(inclH2=true, inclH2O=true),
    each graphite(inclC=true, 'incle-'=true),
    each ionomer(inclC19HF37O5S=true, 'inclH+'=true));

  FCSys.BCs.FaceBus.SubregionFlow bC2[n_y, n_z](
    each gas(
      inclH2O=true,
      inclN2=true,
      inclO2=true,
      O2(redeclare FCSys.BCs.Face.Material.Pressure material, materialSpec(k(
              start=1*U.atm))),
      H2O(redeclare FCSys.BCs.Face.Material.Pressure material, materialSpec(k(
              start=1*U.atm)))),
    each graphite(
      inclC=true,
      'incle-'=true,
      C(redeclare FCSys.BCs.Face.Entropy.Temperature thermal, thermalSpec(k(
              start=298.15*U.K))),
      'e-'(redeclare FCSys.BCs.Face.Material.Current material, redeclare Modelica.Blocks.Sources.Ramp
          materialSpec(duration=1000, height=-2*U.A))),
    each ionomer(
      inclC19HF37O5S=true,
      'inclH+'=true,
      'H+'(redeclare FCSys.BCs.Face.Material.Pressure material, materialSpec(k(
              start=1*U.atm)))));

  inner FCSys.BCs.Environment environment(analysis=true);

equation 
  connect(bC1.face, anCL.xNegative);

  connect(anCL.xPositive, pEM.xNegative);
  connect(pEM.xPositive, caCL.xNegative);

  connect(caCL.xPositive, bC2.face);

end CLToCL;

model CaFP "Test the cathode flow plate"
  import FCSys.Subregions.Phases;
  extends Modelica.Icons.Example;

  parameter Q.Length L_y[:]=fill(1*U.m/1, 1) 
    "Lengths along the channel (Ly)";
  parameter Q.Length L_z[:]=fill(5*U.mm/1, 1) 
    "Lengths across the channel (Lz)";
  final parameter Integer n_y=size(L_y, 1) 
    "Number of regions along the channel";
  final parameter Integer n_z=size(L_z, 1) 
    "Number of regions across the channel";
  inner FCSys.BCs.Environment environment(
    p=149.6*U.kPa,
    T=333.15*U.K,
    analysis=true);
  CaFPs.CaFP caFP(final L_y=L_y, final L_z=L_z);

  FCSys.BCs.FaceBus.SubregionClosed bC1[n_y, n_z](each gas(
      inclH2O=true,
      inclN2=true,
      inclO2=true,
      N2(entropySpec(k=environment.T)),
      O2(entropySpec(k=environment.T)),
      H2O(entropySpec(k=environment.T))),each graphite(
      inclC=true,
      'incle-'=true,
      C(entropySpec(k=environment.T)),
      'e-'(entropySpec(k=environment.T))));
  FCSys.BCs.FaceBus.SubregionClosed bC2[n_y, n_z](each graphite(
      inclC=true,
      'incle-'=true,
      C(entropySpec(k=environment.T)),
      'e-'(entropySpec(k=environment.T),redeclare FCSys.BCs.Face.Material.Pressure
          material)));
  FCSys.BCs.FaceBus.SubregionClosed bC3[caFP.n_x, n_z](each gas(
      inclH2O=true,
      inclN2=true,
      inclO2=true,
      H2O(entropySpec(k=environment.T)),
      N2(entropySpec(k=environment.T)),
      O2(entropySpec(k=environment.T))));
  FCSys.BCs.FaceBus.SubregionClosed bC4[caFP.n_x, n_z](each gas(
      inclH2O=true,
      inclN2=true,
      inclO2=true,
      H2O(entropySpec(k=environment.T)),
      N2(entropySpec(k=environment.T)),
      O2(entropySpec(k=environment.T))));
equation 
  connect(bC1.face, caFP.xNegative);

  connect(caFP.xPositive, bC2.face);

  connect(bC3.face, caFP.yNegative);
  connect(bC4.face, caFP.yPositive);
end CaFP;
