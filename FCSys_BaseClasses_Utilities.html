Modelica definition

function countTrue 
  "Return the number of true entries in a Boolean vector"
  extends Modelica.Icons.Function;

  input Boolean u[:] "Boolean vector";
  output Integer n "Number of true entries";

algorithm 
  n := sum(if u[i] then 1 else 0 for i in 1:size(u, 1));
end countTrue;

FCSys.BaseClasses.Utilities.enumerate

Enumerate the true entries in a Boolean vector (0 for false entries)

TypeNameDescription
Integerenumerated[size(u, 1)]Indices of the true entries (increasing order; 0 for false entries)

Modelica definition

function enumerate 
  "Enumerate the true entries in a Boolean vector (0 for false entries)"
  extends Modelica.Icons.Function;

  input Boolean u[:] "Boolean vector";
  output Integer enumerated[size(u, 1)] 
    "Indices of the true entries (increasing order; 0 for false entries)";

protected 
  Integer count "Counter variable";

algorithm 
  count := 1;
  for i in 1:size(u, 1) loop
    if u[i] then
      enumerated[i] := count;
      count := count + 1;
    else
      enumerated[i] := 0;
    end if;
  end for;
end enumerate;

Modelica definition

function index 
  "Return the indices of the true entries of a Boolean vector"
  extends Modelica.Icons.Function;

  input Boolean u[:] "Boolean array";
  output Integer indices[countTrue(u)] "Indices of the true entries";

protected 
  Integer count "Counter variable";

algorithm 
  count := 1;
  for i in 1:size(u, 1) loop
    if u[i] then
      indices[count] := i;
      count := count + 1;
    end if;
  end for;
end index;
