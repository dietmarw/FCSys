Modelica definition

partial model PartialCondition "Partial model of a condition"
  extends FCSys.Conditions.BaseClasses.Icons.Double;

  parameter Boolean internal=true "Use internal condition";

  replaceable Modelica.Blocks.Sources.Constant source if internal 
    constrainedby Modelica.Blocks.Interfaces.SO "Source of internal condition";

  FCSys.Connectors.RealInput u if not internal "Value of imposed condition";

  FCSys.Connectors.RealOutput y "Expression of measurement";

  FCSys.Connectors.Face negative "Negative-side connector";
  FCSys.Connectors.Face positive "Positive-side connector";

protected 
  FCSys.Connectors.RealOutputInternal u_final 
    "Final value of imposed condition";

equation 
  connect(u, u_final);

  connect(source.y, u_final);
end PartialCondition;
