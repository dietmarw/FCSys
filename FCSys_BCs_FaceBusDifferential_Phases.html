Modelica definition

model Gas "BC for gas"

  extends FCSys.BCs.FaceBusDifferential.Phases.BaseClasses.NullPhase;

  // Conditionally include species.
  parameter Boolean inclH2=false "Hydrogen (H2)";
  FCSys.BCs.FaceDifferential.Species H2 if inclH2 "Model";

  parameter Boolean inclH2O=false "Water (H2O)";
  FCSys.BCs.FaceDifferential.Species H2O if inclH2O "Model";

  parameter Boolean inclN2=false "Nitrogen (N2)";
  FCSys.BCs.FaceDifferential.Species N2 if inclN2 "Model";

  parameter Boolean inclO2=false "Oxygen (O2)";
  FCSys.BCs.FaceDifferential.Species O2 if inclO2 "Model";

equation 
  // H2
  connect(H2.negative, negative.H2);
  connect(H2.positive, positive.H2);
  connect(u.H2, H2.u);

  // H2O
  connect(H2O.negative, negative.H2O);
  connect(H2O.positive, positive.H2O);
  connect(u.H2O, H2O.u);

  // N2
  connect(N2.negative, negative.N2);
  connect(N2.positive, positive.N2);
  connect(u.N2, N2.u);

  // O2
  connect(O2.negative, negative.O2);
  connect(O2.positive, positive.O2);
  connect(u.O2, O2.u);

end Gas;

model Graphite "BC for graphite"

  extends FCSys.BCs.FaceBusDifferential.Phases.BaseClasses.NullPhase;

  // Conditionally include species.
  parameter Boolean inclC=false "Carbon (C)";
  FCSys.BCs.FaceDifferential.Species C(thermoOpt=ThermoOpt.ClosedDiabatic) if 
    inclC "Model";

  parameter Boolean 'incle-'=false "Electrons (e-)";
  FCSys.BCs.FaceDifferential.Species 'e-' if 'incle-' "Model";

equation 
  // C
  connect(C.negative, negative.C);
  connect(C.positive, positive.C);
  connect(u.C, C.u);

  // e-
  connect('e-'.negative, negative.'e-');
  connect('e-'.positive, positive.'e-');
  connect(u.'e-', 'e-'.u);

end Graphite;
