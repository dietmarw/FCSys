Modelica definition

model Species 
  "Condition for a Inert or InertInternal connector (e.g., as in a Species model), with efforts by default"
  extends FCSys.Conditions.BaseClasses.Icons.Single;

  // Included components of linear momentum
  parameter Boolean inclLinX=true "X";
  parameter Boolean inclLinY=false "Y";
  parameter Boolean inclLinZ=false "Z";

  // Included subconnectors
  parameter Boolean inclMechanical=true "Mechanical";
  parameter Boolean inclThermal=true "Thermal";

  // Conditions
  replaceable Mechanical.Velocity mechanicalX(source(k=0)) if inclLinX 
    constrainedby Mechanical.BaseClasses.PartialCondition(
    final inclLinX=inclLinX,
    final inclLinY=inclLinY,
    final inclLinZ=inclLinZ,
    final axis=Axis.x) "X-axis mechanical";
  replaceable Mechanical.Velocity mechanicalY(source(k=0)) if inclLinY 
    constrainedby Mechanical.BaseClasses.PartialCondition(
    final inclLinX=inclLinX,
    final inclLinY=inclLinY,
    final inclLinZ=inclLinZ,
    final axis=Axis.y) "Y-axis mechanical";
  replaceable Mechanical.Velocity mechanicalZ(source(k=0)) if inclLinZ 
    constrainedby Mechanical.BaseClasses.PartialCondition(
    final inclLinX=inclLinX,
    final inclLinY=inclLinY,
    final inclLinZ=inclLinZ,
    final axis=Axis.z) "Z-axis mechanical";
  replaceable Thermal.Temperature thermal(source(k=298.15*U.K)) constrainedby 
    Thermal.BaseClasses.PartialCondition(
    final inclLinX=inclLinX,
    final inclLinY=inclLinY,
    final inclLinZ=inclLinZ) "Thermal";

  FCSys.Connectors.RealInputBus u 
    "Input bus for values of specified conditions";
  FCSys.Connectors.RealOutputBus y "Output bus of measurements";

  FCSys.Connectors.InertInternal inert(
    final n_lin=countTrue({inclLinX,inclLinY,inclLinZ}),
    inclMechanical=inclMechanical,
    inclThermal=inclThermal) 
    "Single-species connector for linear momentum and heat";

equation 
  // X-axis mechanical
  connect(mechanicalX.mechanical, inert.mechanical);
  connect(u.mechanicalX, mechanicalX.u);
  connect(mechanicalX.y, y.mechanicalX);

  // Y-axis mechanical
  connect(mechanicalY.mechanical, inert.mechanical);

  connect(u.mechanicalY, mechanicalY.u);
  connect(mechanicalY.y, y.mechanicalY);

  // Z-axis mechanical
  connect(mechanicalZ.mechanical, inert.mechanical);
  connect(u.mechanicalZ, mechanicalZ.u);
  connect(mechanicalZ.y, y.mechanicalZ);

  // Thermal
  connect(thermal.thermal, inert.thermal);
  connect(u.thermal, thermal.u);
  connect(thermal.y, y.thermal);
end Species;
