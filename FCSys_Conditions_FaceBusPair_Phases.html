Modelica definition

model Gas "Condition for gas"

  extends BaseClasses.NullPhase;

  // Conditionally include species.
  parameter Boolean inclH2=false "Hydrogen (H2)";
  FCSys.Conditions.FacePair.Species H2 if inclH2 "Model";

  parameter Boolean inclH2O=false "Water (H2O)";
  FCSys.Conditions.FacePair.Species H2O if inclH2O "Model";

  parameter Boolean inclN2=false "Nitrogen (N2)";
  FCSys.Conditions.FacePair.Species N2 if inclN2 "Model";

  parameter Boolean inclO2=false "Oxygen (O2)";
  FCSys.Conditions.FacePair.Species O2 if inclO2 "Model";

equation 
  // H2
  connect(H2.negative, negative.H2);
  connect(H2.positive, positive.H2);
  connect(u.H2, H2.u);

  // H2O
  connect(H2O.negative, negative.H2O);
  connect(H2O.positive, positive.H2O);
  connect(u.H2O, H2O.u);

  // N2
  connect(N2.negative, negative.N2);
  connect(N2.positive, positive.N2);
  connect(u.N2, N2.u);

  // O2
  connect(O2.negative, negative.O2);
  connect(O2.positive, positive.O2);
  connect(u.O2, O2.u);

end Gas;

model Ionomer "Condition for ionomer"

  extends BaseClasses.NullPhase;

  // Conditionally include species.
  parameter Boolean 'inclC19HF37O5S-'=false 
    "Nafion sulfonate minus (C19HF37O5S-)";
  FCSys.Conditions.FacePair.Species 'C19HF37O5S-' if 'inclC19HF37O5S-' "Model";

  parameter Boolean 'inclH+'=false "Protons (H+)";
  FCSys.Conditions.FacePair.Species 'H+' if 'inclH+' "Model";

  parameter Boolean inclH2O=false "Water (H2O)";

equation 
  // C19HF37O5S-
  connect('C19HF37O5S-'.negative, negative.'C19HF37O5S-');
  connect('C19HF37O5S-'.positive, positive.'C19HF37O5S-');
  connect(u.'C19HF37O5S-', 'C19HF37O5S-'.u);

  // H2O
