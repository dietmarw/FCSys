FCSys.Phases

Mixtures of species

Information

The graphite, ionomer, and liquid phases can only be used with a compressible phase (gas).

Extends from Modelica.Icons.Package (Icon for standard packages).

Modelica definition

model Gas "Gas phase"
  import FCSys.Utilities.countTrue;
  extends FCSys.Phases.BaseClasses.PartialPhase(final n_spec=countTrue({inclH2,
        inclH2O,inclN2,inclO2}));

  // Conditionally include species.
  parameter Boolean inclH2=false "Include H2";

  replaceable FCSys.Species.H2.Gas.Fixed H2(final n_faces) if inclH2 
    constrainedby FCSys.Species.CompressibleSpecies(
    n_faces=n_faces,
    phi(each stateSelect=if reduceTrans then StateSelect.default else 
          StateSelect.prefer),
    T(stateSelect=if reduceThermal then StateSelect.default else StateSelect.prefer))
    "H2 model";

  parameter Boolean inclH2O=false "Include H2O";

  replaceable FCSys.Species.H2O.Gas.Fixed H2O(final n_faces) if inclH2O 
    constrainedby FCSys.Species.CompressibleSpecies(
    n_faces=n_faces,
    phi(each stateSelect=if reduceTrans then StateSelect.default else 
          StateSelect.prefer),
    T(stateSelect=if reduceThermal then StateSelect.default else StateSelect.prefer))
    "H2O model";

  parameter Boolean inclN2=false "Include N2";

  replaceable FCSys.Species.N2.Gas.Fixed N2(final n_faces) if inclN2 
    constrainedby FCSys.Species.CompressibleSpecies(
    n_faces=n_faces,
    phi(each stateSelect=if reduceTrans then StateSelect.default else 
          StateSelect.prefer),
    T(stateSelect=if reduceThermal then StateSelect.default else StateSelect.prefer))
    "N2 model";

  parameter Boolean inclO2=false "Include O2";

  replaceable FCSys.Species.O2.Gas.Fixed O2(final n_faces) if inclO2 
    constrainedby FCSys.Species.CompressibleSpecies(
    n_faces=n_faces,
    phi(each stateSelect=if reduceTrans then StateSelect.default else 
          StateSelect.prefer),
    T(stateSelect=if reduceThermal then StateSelect.default else StateSelect.prefer))
    "O2 model";

  parameter Boolean inclHOR=false "Hydrogen oxidation";
  parameter Boolean inclORR=false "Oxygen reduction";
  // These can't be outer parameters in Dymola 7.4.

  // Auxiliary variables (for analysis)
  output Q.PressureAbsolute p(stateSelect=StateSelect.never) = amagat.p if 
    n_spec > 0 "Total thermodynamic pressure";

  Connectors.FaceBus xNegative if inclFaces[Axis.x] 
    "Negative face along the x axis";
  Connectors.FaceBus yNegative if inclFaces[Axis.y] 
    "Negative face along the y axis";
  Connectors.FaceBus zNegative if inclFaces[Axis.z] 
    "Negative face along the z axis";
  Connectors.FaceBus xPositive if inclFaces[Axis.x] 
    "Positive face along the x axis";
  Connectors.FaceBus yPositive if inclFaces[Axis.y] 
    "Positive face along the y axis";
  Connectors.FaceBus zPositive if inclFaces[Axis.z] 
    "Positive face along the z axis";
  Connectors.Amagat amagat if n_spec > 0 "Connector for additivity of volume";
  Connectors.Inert inter[n_inter](each final n_trans=n_trans) if n_spec > 0 
    "Connector to exchange momentum and energy with other phases";
  Connectors.Reaction chemical(final n_trans=n_trans) if inclHOR or inclORR 
    "Connector for an electrochemical reaction";
  Connectors.PhysicalBus physical if inclH2O;

  // Aliases
  Q.Velocity phi[n_trans](each stateSelect=StateSelect.prefer) = direct.translational.phi
    if n_spec > 1 and reduceTrans "Velocity";
  Q.Temperature T(each stateSelect=StateSelect.prefer) = direct.thermal.T if 
    n_spec > 1 and reduceThermal "Temperature";
  // These make the selected states more readable.

protected 
  Conditions.Adapters.AmagatDalton DA if n_spec > 0 
    "Adapter between additivity of pressure and additivity of volume";
  Conditions.Adapters.ChemicalReaction HOR(
    final n_trans=n_trans,
    n=-1,
    final m=H2.Data.m) if inclHOR 
    "Adapter for the contribution of H2 to the hydrogen oxidation reaction";
  Conditions.Adapters.ChemicalReaction ORR[2](
    n={-1,2},
    final m={O2.Data.m,H2O.Data.m},
    each final n_trans=n_trans) if inclORR 
    "Adapter for the contribution of O2 and H2O to the oxygen reduction reaction";
  Connectors.DirectInternal direct(
    final n_trans=n_trans,
    final inclTrans=reduceTrans,
    final inclThermal=reduceThermal) if n_spec > 0 and (reduceTrans or 
    reduceThermal) 
    "Connector to directly couple velocities and temperatures within the phase";

equation 
  // Phase change
  connect(H2O.physical, physical.H2O);
  connect(H2O.physical2, physical.H2O2);

  // Reactions
  // ---------
  // HOR
  connect(H2.chemical, HOR.chemical);
  connect(HOR.reaction, chemical);

  // ORR
  connect(O2.chemical, ORR[1].chemical);
  connect(H2O.chemical, ORR[2].chemical);
  connect(ORR[1].reaction, chemical);
  connect(ORR[2].reaction, chemical);

  // Inert exchange
  connect(H2.inter, inter);
  connect(H2O.inter, inter);
  connect(O2.inter, inter);
  connect(N2.inter, inter);
  connect(H2.direct.translational, direct.translational);
  connect(H2.direct.thermal, direct.thermal);
  connect(H2O.direct.translational, direct.translational);
  connect(H2O.direct.thermal, direct.thermal);
  connect(N2.direct.translational, direct.translational);
  connect(N2.direct.thermal, direct.thermal);
  connect(O2.direct.translational, direct.translational);
  connect(O2.direct.thermal, direct.thermal);
  connect(H2.dalton, DA.dalton);
  connect(H2O.dalton, DA.dalton);
  connect(N2.dalton, DA.dalton);
  connect(O2.dalton, DA.dalton);
  connect(DA.amagat, amagat);

  // Transport
  // ---------
  // H2
  connect(H2.faces[facesCart[Axis.x], Side.n], xNegative.H2);

  connect(H2.faces[facesCart[Axis.x], Side.p], xPositive.H2);

  connect(H2.faces[facesCart[Axis.y], Side.n], yNegative.H2);

  connect(H2.faces[facesCart[Axis.y], Side.p], yPositive.H2);

  connect(H2.faces[facesCart[Axis.z], Side.n], zNegative.H2);
  connect(H2.faces[facesCart[Axis.z], Side.p], zPositive.H2);
  // H2O
  connect(H2O.faces[facesCart[Axis.x], Side.n], xNegative.H2O);

  connect(H2O.faces[facesCart[Axis.x], Side.p], xPositive.H2O);

  connect(H2O.faces[facesCart[Axis.y], Side.n], yNegative.H2O);

  connect(H2O.faces[facesCart[Axis.y], Side.p], yPositive.H2O);

  connect(H2O.faces[facesCart[Axis.z], Side.n], zNegative.H2O);
  connect(H2O.faces[facesCart[Axis.z], Side.p], zPositive.H2O);
  // N2
  connect(N2.faces[facesCart[Axis.x], Side.n], xNegative.N2);

  connect(N2.faces[facesCart[Axis.x], Side.p], xPositive.N2);

  connect(N2.faces[facesCart[Axis.y], Side.n], yNegative.N2);

  connect(N2.faces[facesCart[Axis.y], Side.p], yPositive.N2);
  connect(N2.faces[facesCart[Axis.z], Side.n], zNegative.N2);
  connect(N2.faces[facesCart[Axis.z], Side.p], zPositive.N2);

  // O2
  connect(O2.faces[facesCart[Axis.x], Side.n], xNegative.O2);

  connect(O2.faces[facesCart[Axis.x], Side.p], xPositive.O2);

  connect(O2.faces[facesCart[Axis.y], Side.n], yNegative.O2);
  connect(O2.faces[facesCart[Axis.y], Side.p], yPositive.O2);
  connect(O2.faces[facesCart[Axis.z], Side.n], zNegative.O2);
  connect(O2.faces[facesCart[Axis.z], Side.p], zPositive.O2);
end Gas;

model Graphite "Graphite phase"
  import assert = FCSys.Utilities.assertEval;
  import FCSys.Utilities.countTrue;
  extends FCSys.Phases.BaseClasses.PartialPhase(final n_spec=countTrue({
        'inclC+','incle-'}));

  // Conditionally include species.
  parameter Boolean 'inclC+'=false "Include C+";

  replaceable FCSys.Species.'C+'.Graphite.Fixed 'C+'(final n_faces) if 'inclC+'
    constrainedby FCSys.Species.CompressibleSpecies(
    n_faces=n_faces,
    phi(each stateSelect=if reduceTrans then StateSelect.default else 
          StateSelect.prefer),
    T(stateSelect=if reduceThermal then StateSelect.default else StateSelect.prefer))
    "C+ model";

  parameter Boolean 'incle-'=false "Include e-";

  replaceable FCSys.Species.'e-'.Graphite.Fixed 'e-'(final n_faces) if 'incle-'
     and not (inclHOR or inclORR) constrainedby 
    FCSys.Species.CompressibleSpecies(
    n_faces=n_faces,
    phi(each stateSelect=if reduceTrans then StateSelect.default else 
          StateSelect.prefer),
    T(stateSelect=if reduceThermal then StateSelect.default else StateSelect.prefer))
    "e- model";

  //initMaterial=if inclVoid or 'inclC+' then else ,

  // Reaction parameters
  // -------------------
  parameter Boolean inclHOR=false "Include hydrogen oxidation";
  parameter Boolean inclORR=false "Include oxygen reduction";
  parameter Q.CurrentAreicAbsolute J0=1e-7*U.A/U.cm^2 if inclHOR or inclORR 
    "Exchange current density";
  parameter Q.CurrentAreic J_irr=0 if inclHOR or inclORR 
    "Irreversible current of side-reactions";
  // These can't be outer parameters in Dymola 7.4.

  Connectors.FaceBus xNegative if inclFaces[Axis.x] 
    "Negative face along the x axis";
  Connectors.FaceBus yNegative if inclFaces[Axis.y] 
    "Negative face along the y axis";
  Connectors.FaceBus zNegative if inclFaces[Axis.z] 
    "Negative face along the z axis";
  Connectors.FaceBus xPositive if inclFaces[Axis.x] 
    "Positive face along the x axis";
  Connectors.FaceBus yPositive if inclFaces[Axis.y] 
    "Positive face along the y axis";
  Connectors.FaceBus zPositive if inclFaces[Axis.z] 
    "Positive face along the z axis";
  Connectors.Inert inter[n_inter](each final n_trans=n_trans) if n_spec > 0 
    "Connector to exchange momentum and energy with other phases";

  Connectors.Amagat amagat if n_spec > 0 "Connector for additivity of volume";
  Connectors.Reaction chemical(final n_trans=n_trans) if inclHOR or inclORR 
    "Connector for an electrochemical reaction";

  Conditions.Adapters.FaceReaction HOR(
    final axis=Axis.x,
    final side=Side.n,
    final A=A[Axis.x],
    final cartTrans=cartTrans,
    redeclare package Data = FCSys.Characteristics.'e-'.Gas,
    n=2) if inclHOR 
    "Adapter for the contribution of e- to the hydrogen oxidation reaction";

  Conditions.Adapters.FaceReaction ORR(
    final axis=Axis.x,
    final side=Side.p,
    final A=A[Axis.x],
    cartTrans=cartTrans,
    redeclare package Data = FCSys.Characteristics.'e-'.Gas,
    n=-4) if inclORR 
    "Adapter for the contribution of e- to the oxygen reduction reaction";

  // Aliases
  Q.Velocity phi[n_trans](each stateSelect=StateSelect.prefer) = direct.translational.phi
    if n_spec > 1 and reduceTrans "Velocity";
  Q.Temperature T(each stateSelect=StateSelect.prefer) = direct.thermal.T if 
    n_spec > 1 and reduceThermal "Temperature";
  // These make the selected states more readable.

protected 
  Conditions.Adapters.AmagatDalton DA if n_spec > 0 
    "Adapter between additivity of pressure and additivity of volume";
  FCSys.Species.Reaction reaction(
    final A=A[Axis.x],
    transSubstrate=true,
    thermalSubstrate=true,
    fromCurrent=true,
    J0=J0,
    J_irr=J_irr) if inclHOR or inclORR "Model to determine the reaction rate";

  Connectors.DirectInternal direct(
    final n_trans=n_trans,
    final inclTrans=reduceTrans,
    final inclThermal=reduceThermal) if n_spec > 0 and (reduceTrans or 
    reduceThermal) 
    "Connector to directly couple velocities and temperatures within the phase";

equation 
  // Reactions
  // ---------
  connect(reaction.inert, 'C+'.direct);
  // HOR
  connect(HOR.face, xNegative.'e-');

  connect(ORR.face, xPositive.'e-');

  // Inert exchange
  connect('C+'.inter, inter);
  connect('e-'.inter, inter);

  connect('C+'.faces[facesCart[Axis.y], Side.n], yNegative.'C+');

  connect('C+'.faces[facesCart[Axis.y], Side.p], yPositive.'C+');

  connect('C+'.faces[facesCart[Axis.z], Side.n], zNegative.'C+');
  connect('C+'.faces[facesCart[Axis.z], Side.p], zPositive.'C+');

  // e-
  connect('e-'.faces[facesCart[Axis.x], Side.n], xNegative.'e-');

  connect('e-'.faces[facesCart[Axis.x], Side.p], xPositive.'e-');

  connect('e-'.faces[facesCart[Axis.y], Side.n], yNegative.'e-');

  connect('e-'.faces[facesCart[Axis.y], Side.p], yPositive.'e-');

  connect('e-'.faces[facesCart[Axis.z], Side.n], zNegative.'e-');
  connect('e-'.faces[facesCart[Axis.z], Side.p], zPositive.'e-');
  connect(HOR.reaction, reaction.reaction);

  connect(ORR.reaction, reaction.reaction);
  connect(chemical, reaction.reaction);
end Graphite;
