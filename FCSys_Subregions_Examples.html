model Subregion 
  "Evaluate a single subregion, with H2 by default"

  extends Modelica.Icons.Example;

  parameter Boolean 'inclC+'=false "Carbon plus (C+)";
  parameter Boolean 'inclC19HF37O5S-'=false 
    "Nafion sulfonate (C19HF37O5S-)";
  parameter Boolean 'incle-'=false "Electrons (e-)";
  parameter Boolean 'inclH+'=false "Protons (H+)";
  parameter Boolean inclH2=true "Hydrogen (H2)";
  parameter Boolean inclH2O=false "Water vapor (H2O)";
  parameter Boolean inclN2=false "Nitrogen (N2)";
  parameter Boolean inclO2=false "Oxygen (O2)";

  inner Conditions.Environment environment(analysis=true);
  FCSys.Subregions.Subregion subregion(
    L={1,1,1}*U.cm,
    inclTransY=false,
    inclTransZ=false,
    inclFacesY=false,
    inclFacesZ=false,
    inclFacesX=true,
    graphite(
      final 'inclC+'='inclC+',
      final 'incle-'='incle-',
      'C+'(V_IC=subregion.V/4)),
    gas(
      final inclH2=inclH2,
      final inclH2O=inclH2O,
      final inclN2=inclN2,
      final inclO2=inclO2),
    liquid(H2O(V_IC=subregion.V/4)),
    ionomer(
      final 'inclC19HF37O5S-'='inclC19HF37O5S-',
      final 'inclH+'='inclH+',
      'C19HF37O5S-'(V_IC=subregion.V/4)));

end Subregion;

FCSys.Subregions.Examples.SubregionEvaporation

Test a subregion with evaporation and condensation

Information

Initially, the water vapor is below saturation and liquid water is present. Some of the liquid evaporates until saturation is reached. Then at 1 s additional water vapor is injected from the negative boundary and some of the water vapor condenses.

Extends from Examples.Subregion (Evaluate a single subregion, with H2 by default).

Parameters

TypeNameDefaultDescription
Species
Boolean'inclC+'falseCarbon plus (C+)
Boolean'inclC19HF37O5S-'falseNafion sulfonate (C19HF37O5S-)
Boolean'incle-'falseElectrons (e-)
Boolean'inclH+'falseProtons (H+)
BooleaninclH2falseHydrogen (H2)
BooleaninclH2OtrueWater vapor (H2O)
BooleaninclN2falseNitrogen (N2)
BooleaninclO2falseOxygen (O2)

Modelica definition

model SubregionEvaporation 
  "Test a subregion with evaporation and condensation"

  extends Examples.Subregion(
    inclH2O=true,
    inclH2=false,
    subregion(liquid(inclH2O=inclH2O), gas(H2O(p_IC=U.kPa))));
  Conditions.ByConnector.FaceBus.Single.FaceBusIsolated BC1(liquid(inclH2O=true),
      gas(inclH2O=true, H2O(redeclare FCSys.Conditions.ByConnector.Face.Single.Material.Current
          material(redeclare Modelica.Blocks.Sources.Pulse source(
            amplitude=-U.A,
            width=1,
            period=100,
            startTime=1)))));
  Conditions.ByConnector.FaceBus.Single.FaceBusIsolated BC2(gas(inclH2O=true),
      liquid(inclH2O=true));

equation 
  connect(BC1.face, subregion.xNegative);

  connect(subregion.xPositive, BC2.face);

end SubregionEvaporation;

Modelica definition

model SubregionHOR 
  "Test a subregion with the hydrogen oxidation reaction and essential species (C+, C19HF37O5S-, e-, H+, and H2)"

  output Q.Potential w=-subregion.reaction.chemical.mu/2 "Reaction potential";
  output Q.Current zI=subregion.graphite.'e-'.chemical.Ndot 
    "Electrical current due to reaction";
  output Q.Number zJ_Apercm2=zI*U.cm^2/(subregion.A[Axis.x]*U.A) 
    "Electrical current density, in A/cm2";
  output Q.Power Qdot=subregion.graphite.'C+'.Edot_DE "Rate of heat generation";
  output Q.Power P=w*zI "Electrical power";
  output Q.NumberAbsolute eta=P/(P + Qdot) "Efficiency";
  // **Fix Qdot, P, eta

  extends Examples.Subregion(
    'inclC+'=true,
    'inclC19HF37O5S-'=true,
    'incle-'=true,
    'inclH+'=true,
    inclH2=true,
    subregion(
      L={0.287*U.mm,10*U.cm,10*U.cm},
      gas(H2(
          consTransX=Conservation.IC,
          initMaterial=FCSys.Subregions.Species.BaseClasses.InitScalar.None,
          chemical(Ndot(start=0, fixed=true)))),
      graphite(reduceTemp=true,'e-'(
          initMaterial=InitScalar.Amount,
          initTransX=InitTranslational.None,
          phi(each stateSelect=StateSelect.always))),
      ionomer(reduceTemp=true, 'H+'(initTransX=InitTranslational.None)),
      reaction(J_0=0.5*U.A/U.cm^2)));

  Conditions.ByConnector.FaceBus.Single.FaceBusIsolated negativeBC(
    gas(
      final inclH2=inclH2,
      final inclH2O=inclH2O,
      final inclN2=inclN2,
      final inclO2=inclO2,
      H2(redeclare Conditions.ByConnector.Face.Single.Material.Pressure
          material(source(y=environment.p)))),
    graphite(
      'inclC+'=false,
      final 'incle-'='incle-',
      'e-'(redeclare Conditions.ByConnector.Face.Single.Translational.Current
          normal(A=subregion.A[Axis.x], redeclare Modelica.Blocks.Sources.Ramp
            source(height=-150*U.A, duration=100)))),
    ionomer(
      'inclC19HF37O5S-'=false,
      final 'inclH+'='inclH+',
      'H+'(redeclare Conditions.ByConnector.Face.Single.Translational.Force
          normal)));

  Conditions.ByConnector.FaceBus.Single.FaceBusIsolated positiveBC(
    gas(
      final inclH2=inclH2,
      final inclH2O=inclH2O,
      final inclN2=inclN2,
      final inclO2=inclO2),
    graphite(
      'inclC+'=false,
      final 'incle-'='incle-',
      'e-'(redeclare Conditions.ByConnector.Face.Single.Translational.Force
          normal)),
    ionomer(
      'inclC19HF37O5S-'=false,
      final 'inclH+'='inclH+',
      'H+'(redeclare Conditions.ByConnector.Face.Single.Material.Density
          material(source(final y=17842.7*U.C/U.cc)))));
  // **set density properly
equation 
  connect(negativeBC.face, subregion.xNegative);

  connect(subregion.xPositive, positiveBC.face);

end SubregionHOR;

Modelica definition

model SubregionPipeFlow
  import assert = FCSys.BaseClasses.Utilities.assertEval;
  extends Examples.Subregion(subregion(
      L={100,1,1}*U.mm,
      inclFacesY=true,
      inclFacesZ=true));

  parameter Q.Pressure Deltap=-10*U.Pa 
    "Prescribed gas pressure difference (Δp)";

  output Q.Length D=2*subregion.A[Axis.x]/(subregion.L[Axis.y] + subregion.L[
      Axis.z]);
  output Q.Number Re=subregion.gas.H2.phi[Axis.x]*D*subregion.gas.H2.zeta*
      subregion.gas.H2.Data.m*subregion.gas.H2.rho "Reynolds number";
  output Q.Pressure Deltap_Poiseuille=-32*subregion.L[Axis.x]*subregion.gas.H2.phi[
      Axis.x]/(D^2*subregion.gas.H2.zeta) 
    "Pressure difference according to Poiseuille's law";

  Conditions.ByConnector.FaceBus.Single.FaceBusIsolated BC1(
    graphite('inclC+'=false, final 'incle-'='incle-'),
    ionomer('inclC19HF37O5S-'=false, final 'inclH+'='inclH+'),
    gas(
      final inclH2=inclH2,
      final inclH2O=inclH2O,
      final inclN2=inclN2,
      final inclO2=inclO2,
      H2(redeclare Conditions.ByConnector.Face.Single.Translational.Force
          normal, redeclare Conditions.ByConnector.Face.Single.Material.Pressure
          material(source(y=environment.p - Deltap/2))),
      H2O(redeclare Conditions.ByConnector.Face.Single.Translational.Force
          normal, redeclare Conditions.ByConnector.Face.Single.Material.Pressure
          material(source(y=environment.p - Deltap/2))),
      N2(redeclare Conditions.ByConnector.Face.Single.Translational.Force
          normal, redeclare Conditions.ByConnector.Face.Single.Material.Pressure
          material(source(y=environment.p - Deltap/2))),
      O2(redeclare Conditions.ByConnector.Face.Single.Translational.Force
          normal, redeclare Conditions.ByConnector.Face.Single.Material.Pressure
          material(source(y=environment.p - Deltap/2)))));

  Conditions.ByConnector.FaceBus.Single.FaceBusIsolated BC2(
    graphite('inclC+'=false, final 'incle-'='incle-'),
    ionomer('inclC19HF37O5S-'=false, final 'inclH+'='inclH+'),
    gas(
      final inclH2=inclH2,
      final inclH2O=inclH2O,
      final inclN2=inclN2,
      final inclO2=inclO2,
      H2(redeclare Conditions.ByConnector.Face.Single.Translational.Force
          normal, redeclare Conditions.ByConnector.Face.Single.Material.Pressure
          material(source(y=environment.p + Deltap/2))),
      H2O(redeclare Conditions.ByConnector.Face.Single.Translational.Force
          normal, redeclare Conditions.ByConnector.Face.Single.Material.Pressure
          material(source(y=environment.p + Deltap/2))),

  Conditions.ByConnector.FaceBus.Single.FaceBus BC3(

Modelica definition

model Subregions 
  "Test a one-dimensional array of subregions with an initial pressure difference (C+ and H2 included by default)"
  extends Modelica.Icons.Example;

  parameter Integer n_x=0 
    "Number of discrete subregions along the x axis, besides the 2 side subregions (nx)";
  parameter Q.Pressure Deltap_IC=100*U.Pa 
    "Initial pressure difference (ΔpIC)";
  parameter Boolean 'inclC+'=true "Carbon plus (C+)";
  parameter Boolean 'inclC19HF37O5S-'=false 
    "Nafion sulfonate (C19HF37O5S)";
  parameter Boolean 'incle-'=false "Electrons (e-)";
  parameter Boolean inclH2=true "Hydrogen (H2)";
  parameter Boolean inclH2O=false "Water vapor (H2O)";
  parameter Boolean 'inclH+'=false "Protons (H+)";
  parameter Boolean inclN2=false "Nitrogen (N2)";
  parameter Boolean inclO2=false "Oxygen (O2)";

  inner Conditions.Environment environment(analysis=true);
  FCSys.Subregions.Subregion subregion1(
    L={1,1,1}*U.cm,
    inclFacesY=false,
    inclFacesZ=false,
    inclTransY=false,
    inclTransZ=false,
    gas(
      reduceVel=true,
      reduceTemp=true,
      final inclH2=inclH2,
      final inclH2O=inclH2O,
      final inclN2=inclN2,
      final inclO2=inclO2,
      H2O(p_IC=environment.p + Deltap_IC/2),
      N2(p_IC=environment.p + Deltap_IC/2),
      O2(p_IC=environment.p + Deltap_IC/2),
      H2(p_IC=environment.p + Deltap_IC/2)),
    graphite(
      reduceTemp=true,
      final 'inclC+'='inclC+',
      final 'incle-'='incle-',
      'C+'(V_IC=subregion1.V/1000)),
    ionomer(
      reduceTemp=true,
      final 'inclC19HF37O5S-'='inclC19HF37O5S-',
      final 'inclH+'='inclH+',
      'C19HF37O5S-'(V_IC=subregion1.V/1000)),
    liquid(H2O(V_IC=subregion1.V/4)));

  FCSys.Subregions.Subregion subregions[n_x](
    each L={1,1,1}*U.cm,

model ElectricalConduction 
  "Test a one-dimensional array of subregions with C+ and e-"

  extends Examples.Subregion(
    'inclC+'=true,
    'incle-'=true,
    inclH2=false,
    subregion(L={U.cm,U.mm,U.mm},graphite(
        reduceTemp=true,
        'C+'(consMaterial=Conservation.IC, initMaterial=InitScalar.Pressure),
        'e-'(
          mu=0.1*U.mm^2/(U.V*U.s),
          initTransX=FCSys.Subregions.Species.BaseClasses.InitTranslational.None,

          rho_IC=17842.7*U.C/U.cc))));

  // **Integrate the density into the model.

  output Q.Potential w=(subregion.graphite.'e-'.faces[1, Side.p].mPhidot[
      Orientation.normal]/subregion.graphite.'e-'.faces[1, Side.p].rho -
      subregion.graphite.'e-'.faces[1, Side.n].mPhidot[Orientation.normal]/
      subregion.graphite.'e-'.faces[1, Side.n].rho)/subregion.A[Axis.x] 
    "Potential";
  output Q.Current zI=-subregion.graphite.'e-'.Ndot_faces[1, Side.n] 
    "Electrical current";
  output Q.ResistanceElectrical R=w/zI "Electrical resistance as measured";
  output Q.ResistanceElectrical R_Ohm=subregion.graphite.'e-'.v/subregion.graphite.
      'e-'.mu*subregion.L[Axis.x]/subregion.A[Axis.x] 
    "Electrical resistance from Ohm's law";

  Conditions.ByConnector.FaceBus.Single.FaceBusIsolated BC1(
    gas(
      final inclH2=inclH2,
      final inclH2O=inclH2O,
      final inclN2=inclN2,
      final inclO2=inclO2),
    ionomer('inclC19HF37O5S-'=false, final 'inclH+'='inclH+'),
    graphite(
      'inclC+'=false,
      final 'incle-'='incle-',
      'e-'(normal(source(y=U.A)))));
  Conditions.ByConnector.FaceBus.Single.FaceBusIsolated BC2(
    gas(
      final inclH2=inclH2,
      final inclH2O=inclH2O,
      final inclN2=inclN2,
      final inclO2=inclO2),
    ionomer('inclC19HF37O5S-'=false, final 'inclH+'='inclH+'),
    graphite(
      'inclC+'=false,
      final 'incle-'='incle-',
      'e-'(redeclare Conditions.ByConnector.Face.Single.Translational.Force
          normal)));

equation 
  connect(BC1.face, subregion.xNegative);

  connect(subregion.xPositive, BC2.face);

end ElectricalConduction;

model Reaction "Test an electrochemical reaction"
  extends Modelica.Icons.Example;
  import FCSys.BaseClasses.Utilities.cartWrap;
  import FCSys.BaseClasses.Utilities.countTrue;
  import FCSys.BaseClasses.Utilities.enumerate;
  import FCSys.BaseClasses.Utilities.index;

  // Assumptions
  // -----------
  // Included components of translational momentum
  parameter Boolean inclTransX=true "X";
  parameter Boolean inclTransY=false "Y";
  parameter Boolean inclTransZ=false "Z";

  // Auxiliary variables (for analysis)
  output Q.Potential w(stateSelect=StateSelect.never) = reaction.chemical.mu 
    "Electrochemical overpotential";
  output Q.Current zI(stateSelect=StateSelect.never) = -reaction.chemical.Ndot 
    "Electrical current due to reaction";
  output Q.Number zJ_Apercm2(stateSelect=StateSelect.never) = zI*U.cm^2/(
    reaction.A*U.A) "Electrical current density, in A/cm2";
  output Q.Number Qdot[3](each stateSelect=StateSelect.never) = {potential[1].sT_actual
    *potential[1].chemical.Ndot,potential[2].sT_actual*potential[2].chemical.Ndot,
    current.sT_actual*current.chemical.Ndot} "Rates of thermal advection";

  inner Conditions.Environment environment(analysis=true);
  FCSys.Subregions.Reaction reaction;

  ChemicalExchange chemicalExchange[3](n={-2,-2,1}, m={1,1,1}*U.g/U.mol);
  Depletion depletion1(
    transSubstrate=true,
    thermalSubstrate=true,
    side=FCSys.BaseClasses.Side.n,
    redeclare package Data = FCSys.Characteristics.'e-'.Gas);
  Depletion depletion2(side=FCSys.BaseClasses.Side.p, redeclare package Data = FCSys.Characteristics.'H+'.Gas);
  replaceable Conditions.ByConnector.ChemicalSpecies.Current current(
    final inclTransX=inclTransX,
    final inclTransY=inclTransY,
    final inclTransZ=inclTransZ,
    sT=3000*U.K,
    redeclare Modelica.Blocks.Sources.Ramp source(duration=100, height=-100*U.A));

  replaceable Conditions.ByConnector.ChemicalSpecies.Potential potential[2](
    each sT=0,
    each final inclTransX=inclTransX,
    each final inclTransY=inclTransY,
    each final inclTransZ=inclTransZ);
  Conditions.ByConnector.Inert.Inert catalyst(
    thermal(source(y=360*U.K)),
    final inclTransX=inclTransX,
    final inclTransY=inclTransY,
    final inclTransZ=inclTransZ);

  Conditions.ByConnector.Face.Single.Face layer1;
  Conditions.ByConnector.Face.Single.Face layer2;

protected 
  final inner parameter Boolean inclTrans[Axis]={inclTransX,inclTransY,
      inclTransZ} 
    "true, if each component of translational momentum is included";
  final inner parameter Integer n_trans=countTrue(inclTrans) 
    "Number of components of translational momentum";
  final inner parameter Integer cartTrans[:]=index(inclTrans) 
    "Cartesian-axis indices of the components of translational momentum";

equation 
  connect(chemicalExchange[1].reaction, reaction.chemical);
  connect(chemicalExchange[2].reaction, reaction.chemical);
  connect(chemicalExchange[3].reaction, reaction.chemical);
  connect(depletion2.chemical, reaction.chemical);
  connect(depletion1.chemical, reaction.chemical);
  connect(current.chemical, chemicalExchange[3].species);

  connect(potential.chemical, chemicalExchange[1:2].species);
  connect(catalyst.inert, depletion1.inert);
  connect(catalyst.inert, depletion2.inert);
  connect(layer1.face, depletion1.face);
  connect(depletion2.face, layer2.face);
end Reaction;
