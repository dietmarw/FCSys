FCSys.Subregions.Phases

Phases or mixtures of species

Information

The graphite, ionomer, and liquid phases can only be used with a compressible phase (gas).

Extends from Modelica.Icons.Package (Icon for standard packages).

model Gas "Gas phase"
  import FCSys.BaseClasses.Utilities.countTrue;
  extends FCSys.Subregions.Phases.BaseClasses.EmptyPhase(final n_spec=countTrue
        ({inclH2,inclH2O,inclN2,inclO2}));

  // Conditionally include species.
  parameter Boolean inclH2=false "Hydrogen (H2)";

  replaceable Species.H2.Gas.Fixed H2(final n_faces) if inclH2 constrainedby 
    Species.Species(
    n_faces=n_faces,
    initTransX=if initVelX and reduceVel then InitTranslational.None else 
        InitTranslational.Velocity,
    initTransY=if initVelY and reduceVel then InitTranslational.None else 
        InitTranslational.Velocity,
    initTransZ=if initVelZ and reduceVel then InitTranslational.None else 
        InitTranslational.Velocity,
    initEnergy=if initTemp and reduceTemp then InitScalar.None else InitScalar.Temperature,

    phi(each stateSelect=if reduceVel then StateSelect.default else StateSelect.prefer),

    T(stateSelect=if reduceTemp then StateSelect.default else StateSelect.prefer))
    "H2 model";

  parameter Boolean inclH2O=true "Water (H2O)";
  replaceable Species.H2O.Gas.Fixed H2O(final n_faces) if inclH2O 
    constrainedby Species.Species(
    n_faces=n_faces,
    initTransX=if initVelX and reduceVel then InitTranslational.None else 
        InitTranslational.Velocity,
    initTransY=if initVelY and reduceVel then InitTranslational.None else 
        InitTranslational.Velocity,
    initTransZ=if initVelZ and reduceVel then InitTranslational.None else 
        InitTranslational.Velocity,
    initEnergy=if initTemp and reduceTemp then InitScalar.None else InitScalar.Temperature,

    phi(each stateSelect=if reduceVel then StateSelect.default else StateSelect.prefer),

    T(stateSelect=if reduceTemp then StateSelect.default else StateSelect.prefer))
    "H2O model";

  parameter Boolean inclN2=false "Nitrogen (N2)";
  replaceable Species.N2.Gas.Fixed N2(final n_faces) if inclN2 constrainedby 
    Species.Species(
    n_faces=n_faces,
    initTransX=if initVelX and reduceVel then InitTranslational.None else 
        InitTranslational.Velocity,
    initTransY=if initVelY and reduceVel then InitTranslational.None else 
        InitTranslational.Velocity,
    initTransZ=if initVelZ and reduceVel then InitTranslational.None else 
        InitTranslational.Velocity,
    initEnergy=if initTemp and reduceTemp then InitScalar.None else InitScalar.Temperature,

    phi(each stateSelect=if reduceVel then StateSelect.default else StateSelect.prefer),

    T(stateSelect=if reduceTemp then StateSelect.default else StateSelect.prefer))
    "N2 model";

  parameter Boolean inclO2=false "Oxygen (O2)";
  replaceable Species.O2.Gas.Fixed O2(final n_faces) if inclO2 constrainedby 
    Species.Species(
    n_faces=n_faces,
    initTransX=if initVelX and reduceVel then InitTranslational.None else 
        InitTranslational.Velocity,
    initTransY=if initVelY and reduceVel then InitTranslational.None else 
        InitTranslational.Velocity,
    initTransZ=if initVelZ and reduceVel then InitTranslational.None else 
        InitTranslational.Velocity,
    initEnergy=if initTemp and reduceTemp then InitScalar.None else InitScalar.Temperature,

    phi(each stateSelect=if reduceVel then StateSelect.default else StateSelect.prefer),

    T(stateSelect=if reduceTemp then StateSelect.default else StateSelect.prefer))
    "O2 model";

  ChemicalExchange HOR(n=1, final m=H2.Data.m) if inclHOR 
    "Interface with the hydrogen oxidation reaction";
  ChemicalExchange ORR[2](n={-1,2}, final m={O2.Data.m,H2O.Data.m}) if inclORR 
    "Interfaces with the oxygen reduction reaction";
  Connectors.ChemicalReaction chemical if inclHOR or inclORR 
    "Connector for chemical reactions";
  Connectors.PhysicalBus physical if inclH2O "Connector for phase change";

  parameter Boolean inclHOR=false "Hydrogen oxidation";
  parameter Boolean inclORR=false "Oxygen reduction";
  // These can't be outer parameters in Dymola 7.4.

equation 
  // Phase change
  connect(physical.H2O, H2O.physical);

  // Reactions
  // ---------
  // HOR
  connect(H2.chemical, HOR.species);
  connect(HOR.reaction, chemical);
  // ORR
  connect(O2.chemical, ORR[1].species);
  connect(H2O.chemical, ORR[2].species);
  connect(ORR[1].reaction, chemical);
  connect(ORR[2].reaction, chemical);

  // H2
  // --
  // Diffusive exchange
  connect(H2.inert.translational, common.translational);
  connect(H2.inert.thermal, common.thermal);
  connect(H2.inertDalton, inertDalton);
  // Transport
  connect(H2.faces[facesCart[Axis.x], Side.n], xNegative.H2);

  connect(H2.faces[facesCart[Axis.x], Side.p], xPositive.H2);

  connect(H2.faces[facesCart[Axis.y], Side.n], yNegative.H2);

  connect(H2.faces[facesCart[Axis.y], Side.p], yPositive.H2);

  connect(H2.faces[facesCart[Axis.z], Side.n], zNegative.H2);
  connect(H2.faces[facesCart[Axis.z], Side.p], zPositive.H2);
  // H2O
  // ---
  // Diffusive exchange
  connect(H2O.inert.translational, common.translational);
  connect(H2O.inert.thermal, common.thermal);
  connect(H2O.inertDalton, inertDalton);
  // Transport
  connect(H2O.faces[facesCart[Axis.x], Side.n], xNegative.H2O);

  connect(H2O.faces[facesCart[Axis.x], Side.p], xPositive.H2O);

  connect(H2O.faces[facesCart[Axis.y], Side.n], yNegative.H2O);

  connect(H2O.faces[facesCart[Axis.y], Side.p], yPositive.H2O);

  connect(H2O.faces[facesCart[Axis.z], Side.n], zNegative.H2O);
  connect(H2O.faces[facesCart[Axis.z], Side.p], zPositive.H2O);
  // N2
  // --
  // Diffusive exchange
  connect(N2.inert.translational, common.translational);
  connect(N2.inert.thermal, common.thermal);
  connect(N2.inertDalton, inertDalton);

  // Transport
  connect(N2.faces[facesCart[Axis.x], Side.n], xNegative.N2);
  connect(N2.faces[facesCart[Axis.x], Side.p], xPositive.N2);

  connect(N2.faces[facesCart[Axis.y], Side.n], yNegative.N2);

  connect(N2.faces[facesCart[Axis.y], Side.p], yPositive.N2);
  connect(N2.faces[facesCart[Axis.z], Side.n], zNegative.N2);
  connect(N2.faces[facesCart[Axis.z], Side.p], zPositive.N2);
  // O2
  // --
  // Diffusive exchange
  connect(O2.inert.translational, common.translational);
  connect(O2.inert.thermal, common.thermal);
  connect(O2.inertDalton, inertDalton);

  // Transport
  connect(O2.faces[facesCart[Axis.x], Side.n], xNegative.O2);
  connect(O2.faces[facesCart[Axis.x], Side.p], xPositive.O2);

  connect(O2.faces[facesCart[Axis.y], Side.n], yNegative.O2);
  connect(O2.faces[facesCart[Axis.y], Side.p], yPositive.O2);
  connect(O2.faces[facesCart[Axis.z], Side.n], zNegative.O2);
  connect(O2.faces[facesCart[Axis.z], Side.p], zPositive.O2);
end Gas;

model Graphite "Graphite phase"
  import assert = FCSys.BaseClasses.Utilities.assertEval;
  import FCSys.BaseClasses.Utilities.countTrue;
  extends FCSys.Subregions.Phases.BaseClasses.EmptyPhase(
    final n_spec=countTrue({'inclC+','incle-'}),
    initVelX=not 'inclC+',
    initVelY=not 'inclC+',
    initVelZ=not 'inclC+');

  // Conditionally include species.
  parameter Boolean 'inclC+'='incle-' 
    "Carbon plus (C+)";

  replaceable Species.'C+'.Graphite.Fixed 'C+'(final n_faces) if 'inclC+' 
    constrainedby Species.Species(
    n_faces=n_faces,
    initEnergy=if initTemp and reduceTemp then InitScalar.None else InitScalar.Temperature,

    phi(each stateSelect=if reduceVel then StateSelect.default else StateSelect.prefer),

    T(stateSelect=if reduceTemp then StateSelect.default else StateSelect.prefer))
    "C+ model";

  parameter Boolean 'incle-'=false "Electrons (e-)";

  replaceable Species.'e-'.Graphite.Fixed 'e-'(final n_faces) if 'incle-' 
    constrainedby Species.Species(
    n_faces=n_faces,
    initMaterial=if 'inclC+' then InitScalar.None else InitScalar.Volume,
    initTransX=if (initVelX or 'inclC+') and reduceVel then InitTranslational.None
         else InitTranslational.Velocity,
    initTransY=if (initVelY or 'inclC+') and reduceVel then InitTranslational.None
         else InitTranslational.Velocity,
    initTransZ=if (initVelZ or 'inclC+') and reduceVel then InitTranslational.None
         else InitTranslational.Velocity,
    initEnergy=if initTemp and reduceTemp then InitScalar.None else InitScalar.Temperature,

    phi(each stateSelect=if reduceVel then StateSelect.default else StateSelect.prefer),

    T(stateSelect=if reduceTemp then StateSelect.default else StateSelect.prefer))
    "'e-' model";

  ChemicalExchange HOR(n=-2, final m='e-'.Data.m) if inclHOR 
    "Interface with the hydrogen oxidation reaction";
  ChemicalExchange ORR(n=-4, final m='e-'.Data.m) if inclORR 
    "Interface with the oxygen reduction reaction";
  Connectors.ChemicalReaction chemical if inclHOR or inclORR 
    "Connector for chemical reactions";
  Depletion depletionHOR(
    A=A[Axis.x],
    transSubstrate=true,
    thermalSubstrate=true,
    side=FCSys.BaseClasses.Side.p,
    redeclare package Data = FCSys.Characteristics.'e-'.Gas) if inclHOR 
    "Depletion region for the HOR";

  Depletion depletionORR(
    A=A[Axis.x],
    transSubstrate=true,
    thermalSubstrate=true,
    side=FCSys.BaseClasses.Side.n,
    redeclare package Data = FCSys.Characteristics.'e-'.Gas) if inclORR 
    "Depletion region for the ORR";

  parameter Boolean inclHOR=false "Hydrogen oxidation";
  parameter Boolean inclORR=false "Oxygen reduction";
  // These can't be outer parameters in Dymola 7.4.

initial equation 
  assert(not (inclHOR and inclORR), "The HOR and ORR cannot be included simultaneously.");

equation 
  // Reactions
  // ---------
  // HOR
  connect('e-'.chemical, HOR.species);
  connect(HOR.reaction, chemical);
  connect(depletionHOR.face, 'e-'.faces[1, Side.p]);
  connect(depletionHOR.inert, 'C+'.inert);
  connect(HOR.reaction, chemical);
  // ORR
  connect('e-'.chemical, ORR.species);
  connect(ORR.reaction, chemical);
  connect(depletionORR.face, 'e-'.faces[1, Side.n]);

  connect(depletionORR.inert, 'C+'.inert);
  connect(ORR.reaction, chemical);

  // C+
  // --
  // Exchange
  connect('C+'.inert.translational, common.translational);
  connect('C+'.inert.thermal, common.thermal);
  connect('C+'.inertDalton, inertDalton);
  // Transport
  connect('C+'.faces[facesCart[Axis.x], Side.n], xNegative.'C+');

  connect('C+'.faces[facesCart[Axis.x], Side.p], xPositive.'C+');

  connect('C+'.faces[facesCart[Axis.y], Side.n], yNegative.'C+');

  connect('C+'.faces[facesCart[Axis.y], Side.p], yPositive.'C+');

  connect('C+'.faces[facesCart[Axis.z], Side.n], zNegative.'C+');
  connect('C+'.faces[facesCart[Axis.z], Side.p], zPositive.'C+');

  // e-
  // --
  // Exchange
  connect('e-'.inert.translational, common.translational);
  connect('e-'.inert.thermal, common.thermal);
  connect('e-'.inertDalton, inertDalton);
  // Transport
  connect('e-'.faces[facesCart[Axis.x], Side.n], xNegative.'e-');

  connect('e-'.faces[facesCart[Axis.x], Side.p], xPositive.'e-');

  connect('e-'.faces[facesCart[Axis.y], Side.n], yNegative.'e-');

  connect('e-'.faces[facesCart[Axis.y], Side.p], yPositive.'e-');

  connect('e-'.faces[facesCart[Axis.z], Side.n], zNegative.'e-');
  connect('e-'.faces[facesCart[Axis.z], Side.p], zPositive.'e-');
  connect(depletionORR.chemical, chemical);

  connect(depletionHOR.chemical, chemical);
end Graphite;

model Liquid "Liquid phase"
  extends FCSys.Subregions.Phases.BaseClasses.EmptyPhase(final n_spec=if 
        inclH2O then 1 else 0);

  // Conditionally include species.
  parameter Boolean inclH2O=false "Water (H2O)";
  replaceable Species.H2O.Liquid.Fixed H2O(final n_faces) if inclH2O 
    constrainedby Species.Species(
    n_faces=n_faces,
    initTransX=if initVelX and reduceVel then InitTranslational.None else 
        InitTranslational.Velocity,
    initTransY=if initVelY and reduceVel then InitTranslational.None else 
        InitTranslational.Velocity,
    initTransZ=if initVelZ and reduceVel then InitTranslational.None else 
        InitTranslational.Velocity,
    initEnergy=if initTemp and reduceTemp then InitScalar.None else InitScalar.Temperature,

    phi(each stateSelect=if reduceVel then StateSelect.default else StateSelect.prefer),

    T(stateSelect=if reduceTemp then StateSelect.default else StateSelect.prefer))
    "H2O model";

  Connectors.PhysicalBus physical if inclH2O "Connector for phase change";

equation 
  // Phase change
  connect(physical.H2O, H2O.physical);

  // H2O
  // ---
  // Exchange
  connect(H2O.inert.translational, common.translational);
  connect(H2O.inert.thermal, common.thermal);
  connect(H2O.inertDalton, inertDalton);
  // Transport
  connect(H2O.faces[Axis.x, Side.n], xNegative.H2O);

  connect(H2O.faces[Axis.x, Side.p], xPositive.H2O);

  connect(H2O.faces[facesCart[Axis.y], Side.n], yNegative.H2O);

  connect(H2O.faces[facesCart[Axis.y], Side.p], yPositive.H2O);

  connect(H2O.faces[facesCart[Axis.z], Side.n], zNegative.H2O);
  connect(H2O.faces[facesCart[Axis.z], Side.p], zPositive.H2O);
end Liquid;
