Modelica definition

model Gas "Condition for gas"

  extends BaseClasses.EmptyPhase;

  // Conditionally include species.
  parameter Boolean inclH2=false "Hydrogen (H2)";

  Face.Single.Face H2 if inclH2 "Conditions";

  parameter Boolean inclH2O=false "Water (H2O)";

  Face.Single.Face H2O if inclH2O "Conditions";

  parameter Boolean inclN2=false "Nitrogen (N2)";

  Face.Single.Face N2 if inclN2 "Conditions";

  parameter Boolean inclO2=false "Oxygen (O2)";

  Face.Single.Face O2 if inclO2 "Conditions";

equation 
  // H2
  connect(H2.face, face.H2);
  connect(u.H2, H2.u);

  connect(H2.y, y.H2);

  // H2O
  connect(H2O.face, face.H2O);
  connect(u.H2O, H2O.u);

  connect(H2O.y, y.H2O);

  // N2
  connect(N2.face, face.N2);
  connect(u.N2, N2.u);

  connect(N2.y, y.N2);

  // O2
  connect(O2.face, face.O2);
  connect(u.O2, O2.u);

  connect(O2.y, y.O2);
end Gas;

model Ionomer "Condition for ionomer"

  extends BaseClasses.EmptyPhase;

  // Conditionally include species.
  parameter Boolean 'inclC19HF37O5S-'=false 
    "Nafion sulfonate minus (C19HF37O5S-)";

  Face.Single.Face 'C19HF37O5S-' if 'inclC19HF37O5S-' "Conditions";

  parameter Boolean 'inclH+'=false "Protons (H+)";

  Face.Single.Face 'H+' if 'inclH+' "Conditions";

  parameter Boolean inclH2O=false "Water (H2O)";

  Face.Single.Face H2O if inclH2O "Conditions";

equation 
  // C19HF37O5S-
  connect('C19HF37O5S-'.face, face.'C19HF37O5S-');
  connect(u.'C19HF37O5S-', 'C19HF37O5S-'.u);

  connect('C19HF37O5S-'.y, y.'C19HF37O5S-');

  // H+

  connect(u.'H+', 'H+'.u);

  connect('H+'.y, y.'H+');

end Ionomer;
