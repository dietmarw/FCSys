model Gas "BC for gas"

  extends BaseClasses.NullPhase;

  // Conditionally include species.
  parameter Boolean inclH2=false "Hydrogen (H2)";
  FCSys.BCs.Face.Species H2(final axis=axis) if inclH2 "Model";

  parameter Boolean inclH2O=false "Water (H2O)";
  FCSys.BCs.Face.Species H2O(final axis=axis) if inclH2O "Model";

  parameter Boolean inclN2=false "Nitrogen (N2)";
  FCSys.BCs.Face.Species N2(final axis=axis) if inclN2 "Model";

  parameter Boolean inclO2=false "Oxygen (O2)";
  FCSys.BCs.Face.Species O2(final axis=axis) if inclO2 "Model";

equation 
  // H2
  connect(H2.face.material, face.H2.material);
  connect(H2.face.mechanicalX, face.H2.mechanicalX);
  connect(H2.face.mechanicalY, face.H2.mechanicalY);
  connect(H2.face.mechanicalZ, face.H2.mechanicalZ);
  connect(H2.face.thermal, face.H2.thermal);
  connect(u.H2, H2.u);

  // H2O
  connect(H2O.face.material, face.H2O.material);
  connect(H2O.face.mechanicalX, face.H2O.mechanicalX);
  connect(H2O.face.mechanicalY, face.H2O.mechanicalY);
  connect(H2O.face.mechanicalZ, face.H2O.mechanicalZ);
  connect(H2O.face.thermal, face.H2O.thermal);
  connect(u.H2O, H2O.u);

  // N2
  connect(N2.face.material, face.N2.material);
  connect(N2.face.mechanicalX, face.N2.mechanicalX);
  connect(N2.face.mechanicalY, face.N2.mechanicalY);
  connect(N2.face.mechanicalZ, face.N2.mechanicalZ);
  connect(N2.face.thermal, face.N2.thermal);
  connect(u.N2, N2.u);

  // O2
  connect(O2.face.material, face.O2.material);
  connect(O2.face.mechanicalX, face.O2.mechanicalX);
  connect(O2.face.mechanicalY, face.O2.mechanicalY);
  connect(O2.face.mechanicalZ, face.O2.mechanicalZ);
  connect(O2.face.thermal, face.O2.thermal);
  connect(u.O2, O2.u);

end Gas;

model Graphite "BC for graphite"

  extends BaseClasses.NullPhase;

  // Conditionally include species.
  parameter Boolean inclC=false "Carbon (C)";
  FCSys.BCs.Face.Species C(final axis=axis,thermoOpt=ThermoOpt.ClosedDiabatic) 
    if inclC "Model";

  parameter Boolean 'incle-'=false "Electrons (e-)";
  FCSys.BCs.Face.Species 'e-'(final axis=axis) if 'incle-' "Model";

equation 
  // C
  connect(C.face.material, face.C.material);
  connect(C.face.mechanicalX, face.C.mechanicalX);
  connect(C.face.mechanicalY, face.C.mechanicalY);
  connect(C.face.mechanicalZ, face.C.mechanicalZ);
  connect(C.face.thermal, face.C.thermal);
  connect(u.C, C.u);

  // e-
  connect('e-'.face.material, face.'e-'.material);
  connect('e-'.face.mechanicalX, face.'e-'.mechanicalX);
  connect('e-'.face.mechanicalY, face.'e-'.mechanicalY);
  connect('e-'.face.mechanicalZ, face.'e-'.mechanicalZ);
  connect('e-'.face.thermal, face.H2.thermal);
  connect(u.'e-', 'e-'.u);

end Graphite;
