Modelica definition

partial model PartialTestStand "Partial cell test stand"
  extends FCSys.BaseClasses.Icons.Names.Top9;
  extends Modelica.Icons.UnderConstruction;
  final parameter Integer n_x_an=1 
    "Number of subregions along the through-cell axis in anode FP (nx an)";
  final parameter Integer n_x_ca=1 
    "Number of subregions along the through-cell axis in anode FP (nx ca)";
  final parameter Integer n_y=1 
    "Number of subregions along the channel (ny)";
  final parameter Integer n_z=1 
    "Number of subregions across the channel (nz)";

  parameter Boolean inclIO=false "true, if input and output are included";

  FCSys.Connectors.FaceBus anEnd[n_y, n_z] "Anode end plate";
  FCSys.Connectors.FaceBus caEnd[n_y, n_z] "Cathode end plate";
  FCSys.Connectors.FaceBus anSource[n_x_an, n_z] "Anode source";
  FCSys.Connectors.FaceBus anSink[n_x_an, n_z] "Anode sink";
  FCSys.Connectors.FaceBus caSource[n_x_ca, n_z] "Cathode source";
  FCSys.Connectors.FaceBus caSink[n_x_ca, n_z] "Cathode sink";

  FaceBus.SubregionFlow anEndCondition[n_y, n_z](each graphite('inclC+'=true,
        'incle-'=true));
  FaceBus.SubregionFlow caEndCondition[n_y, n_z](each graphite('inclC+'=true,
        'incle-'=true));
  FaceBus.SubregionFlow anSourceCondition[n_x_an, n_z](each gas(inclH2=true,
        inclH2O=true));
  FaceBus.SubregionFlow anSinkCondition[n_x_an, n_z](each gas(inclH2=true,
        inclH2O=true));
  FaceBus.SubregionFlow caSourceCondition[n_x_ca, n_z](each gas(
      inclH2O=true,
      inclN2=true,
      inclO2=true));
  FaceBus.SubregionFlow caSinkCondition[n_x_ca, n_z](each gas(
      inclH2O=true,
      inclN2=true,
      inclO2=true));
  FCSys.Connectors.RealInputBus u[n_y, n_z] if inclIO;
  FCSys.Connectors.RealOutputBus y[n_y, n_z] if inclIO;
  replaceable FCSys.Conditions.FaceBusPair.Subregion current[n_y, n_z](graphite(
        'inclC+'=true, 'incle-'=true)) if inclIO constrainedby 
    FCSys.Conditions.FaceBusPair.Subregion(graphite('inclC+'=true, 'incle-'=true));

equation 
  connect(anSourceCondition.face, anSource);

  connect(anSinkCondition.face, anSink);

  connect(caSourceCondition.face, caSource);

  connect(caSinkCondition.face, caSink);

  connect(caEndCondition.face, caEnd);
  connect(current.positive, caEnd);

  connect(current.negative, anEnd);

  connect(u, current.u);
  connect(voltage.negative, anEndCondition.face);

  connect(caEnd, voltage.positive);
  connect(voltage.y, y);

  connect(anEndCondition.face, anEnd);
end PartialTestStand;

FCSys.Conditions.TestStands.BaseClasses.PartialTestStandNoIO

Partial cell test stand without inputs/outputs

Information

Extends from FCSys.BaseClasses.Icons.Names.Top9.

Modelica definition

partial model PartialTestStandNoIO 
  "Partial cell test stand without inputs/outputs"
  extends FCSys.BaseClasses.Icons.Names.Top9;

  final parameter Integer n_x_an=1 
    "Number of subregions along the through-cell axis in anode FP (nx an)";
  final parameter Integer n_x_ca=1 
    "Number of subregions along the through-cell axis in anode FP (nx ca)";
  final parameter Integer n_y=1 
    "Number of subregions along the channel (ny)";
  final parameter Integer n_z=1 
    "Number of subregions across the channel (nz)";

  FaceBus.SubregionFlow anEnd[n_y, n_z](each graphite(
      'inclC+'=true,
      'incle-'=true,
      'e-'(redeclare Face.Normal.CurrentAreic normal(redeclare Modelica.Blocks.Sources.Ramp
            spec(height=U.A/U.cm^2, duration=50)))));
  FaceBus.SubregionFlow caEnd[n_y, n_z](each graphite(
      'inclC+'=true,
      'incle-'=true,
      'e-'(redeclare Face.Normal.CurrentAreic normal(redeclare Modelica.Blocks.Sources.Ramp
            spec(height=U.A/U.cm^2, duration=50)))));
  FaceBus.SubregionFlow anSource[n_x_an, n_z](each gas(inclH2=true, inclH2O=true));
  FaceBus.SubregionFlow anSink[n_x_an, n_z](each gas(inclH2=true, inclH2O=true));
  FaceBus.SubregionFlow caSource[n_x_ca, n_z](each gas(
      inclH2O=true,
      inclN2=true,
      inclO2=true));
  FaceBus.SubregionFlow caSink[n_x_ca, n_z](each gas(
      inclH2O=true,
      inclN2=true,
      inclO2=true));

  inner Environment environment;

end PartialTestStandNoIO;
